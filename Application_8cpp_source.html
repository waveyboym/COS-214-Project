127void Application::runApp()
128{
129 CROW_ROUTE(this->app, "/")
130 .websocket()
131 .onopen([&](crow::websocket::connection& conn){
132 CROW_LOG_INFO << color::format_colour::make_colour(color::GREEN) << "new websocket connection" << color::format_colour::make_colour(color::DEFAULT);
133
134 //mutex locking here to prevent multiple client connections from trying to add at the same time
135 std::lock_guard<std::mutex> lock(this->mtx);
136 std::string id = this->insertClient(&conn);
137 conn.send_text("{\"status\":\"success\", \"message\":\"create_token\", \"token\":\"" + id + "\"}");
138 //release lock
139 })
140 .onclose([&](crow::websocket::connection& conn, const std::string& reason){
141 CROW_LOG_INFO << color::format_colour::make_colour(color::RED) << "websocket connection closed: " << reason << color::format_colour::make_colour(color::DEFAULT);
142
143 //mutex locking here to prevent multiple client connections from trying to erase at the same time
144 std::lock_guard<std::mutex> lock(this->mtx);
145 bool remove_status = this->removeClient(&conn);
146
147 if(remove_status){
148 CROW_LOG_INFO << color::format_colour::make_colour(color::GREEN) << "successfully removed client" << color::format_colour::make_colour(color::DEFAULT);
149 }
150 else{
151 CROW_LOG_INFO << color::format_colour::make_colour(color::RED) << "failed to remove client" << color::format_colour::make_colour(color::DEFAULT);
152 }
153 //release lock
154 })
155 .onmessage([&](crow::websocket::connection& conn, const std::string& data, bool is_binary){
156 std::cout << color::format_colour::make_colour(color::BLUE) << "✔ Received message" << color::format_colour::make_colour(color::DEFAULT) << std::endl;
157
158 this->pause_app_execution = true;
159
160 if (is_binary){
161 //mutex locking here to prevent multiple client connections from trying to erase at the same time
162 std::lock_guard<std::mutex> lock(this->mtx);
163 //pause game execution
164 conn.send_binary(data);
165 //release lock
166 }
167 else{
168 //mutex locking here to prevent multiple client connections from trying to erase at the same time
169 std::lock_guard<std::mutex> lock(this->mtx);
170 std::string res = this->processFrontendRequest(data);
171 conn.send_text(res);
172 //release lock
173 std::cout << color::format_colour::make_colour(color::BLUE) << "✔ Sent back response" << color::format_colour::make_colour(color::DEFAULT) << std::endl;
174 }
175
176 this->pause_app_execution = false;
177 });
178
179 this->app.port(18080)
180 //.multithreaded() //if you use all threads, none might be left to execute the game hence why we commented this out
181 .run();
182
183 this->quit_app = true;
184}
