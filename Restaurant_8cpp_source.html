76std::string Restaurant::generateUUID(){
77 std::random_device rd;
78 auto seed_data = std::array<int, std::mt19937::state_size> {};
79 std::generate(std::begin(seed_data), std::end(seed_data), std::ref(rd));
80 std::seed_seq seq(std::begin(seed_data), std::end(seed_data));
81 std::mt19937 generator(seq);
82 uuids::uuid_random_generator gen{generator};
83
84 uuids::uuid const idUUID = gen();
85
86 std::string id = uuids::to_string(idUUID);
87
88 return id;
89}

91void Restaurant::progressByOneStep(){
92 int random_number = rand();
93
94 if(random_number % 10 >= 0 && random_number % 10 <= 5){
95 //0 to 10 customers should enter the restaurant
96 this->customersEnterRestaurant(random_number);
97 }
98 if(random_number % 10 >= 0 && random_number % 10 <= 10){
99 //seat customers who are not seated at an available table
100 this->seatAnyCustomer(random_number);
101 }
102 if(random_number % 10 >= 0 && random_number % 10 <=10){
103 //assign a free waiter to any table that has customers
104 this->assignOneFreeWaiter();
105 }
106 if(random_number % 10 >= 0 && random_number % 10 <=1){
107 //assign all free waiters to any table that has customers
108 this->assignAllFreeWaiters();
109 }
110 if(random_number % 10 >= 0 && random_number % 10 < 7)
111 {
112 //0 to 6 customers will decide what to order
113 setAnyCustomerOrder(random_number);
114 }
115 if(random_number % 10 >= 0 && random_number % 10 < 6)
116 {
117 //waiter will take orders
118 waiterTakesOrder();
119 }
120 if(random_number % 10 >= 0 && random_number % 10 < 8)
121 {
122 //waiter will send orders to the kitchen to prepare
123 ordersTakenToKitchen();
124 }
125 if(random_number % 10 >= 0 && random_number % 10 < 6)
126 {
127 //kitchen will tell waiters to take completed meals to customers
128 deliverMeals();
129 }
130 if(random_number % 10 >= 0 && random_number % 10 <= 10){
131 //assign a free waiter to any table that has customers
132 this->unseatFinishedCustomers();
133 }
134}
